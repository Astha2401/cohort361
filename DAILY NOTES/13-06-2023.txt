#!/bin/bash
armstrong=0
copy1=$1
copy2=$1
digits=0
#find the length of the number
while (( $copy1!=0 ))
do
copy1=$(( $copy1 / 10 ))
digits=$(( $digits + 1 ))
done
#find armstrong sum
while (( $copy2!=0 ))
do
lastdigit=$(( $copy2 % 10 ))
armstrong=$(( $armstrong + ($lastdigit**$digits) ))
copy2=$(( $copy2 / 10 ))
done
#check if armstrong
if (( $armstrong==$1 ))
then
echo "Armstrong"
else
echo "Not Armstrong"
fi
